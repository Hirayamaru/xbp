第４回課題

今回は数字当てゲームを作成しました。こちらがpythonのコードになります。

  1. import random
  3. def main():
  4.     lower_bound = 1
  5.     upper_bound = 100
  6.     target_number = random.randint(lower_bound, upper_bound) #1から100までの数字がランダムで選ばれるようになった。
  7.     attempts = 0 #attemptsで何回挑戦したのかをカウントできる。
  9.     print(f"1から100までの数で数字を当ててください。")
  10.     print(f"ヒント: {lower_bound}から{upper_bound}の範囲です。") #1から100までの数字を入力してもらうようにprintで文を出す。
  12.     while True:
  13.         try:
  14.             guess = int(input("予想した数を入力して下さい："))
  15.             attempts += 1
  17.             if guess < lower_bound or guess > upper_bound: #予想の数字が大きいか小さいか。
  18.                 print(f"有効な範囲内の数字を入力してください({lower_bound}から{upper_bound}の間)。")
  19.             elif guess < target_number:
  20.                 print("もっと大きな数字です。") #入力した数字より正解の数字が大きいときに出る。
  21.             elif guess > target_number:
  22.                 print("もっと小さな数字です。") #入力した数字より正解の数字が小さいときに出る。
  23.             else:
  24.                 print(f"正解！{target_number}を{attempts}回目で当てました。") #正解の数字をtarget_numberで出し、回答回数
  25.                 break # 正解した場合はループを終了します。
  26.         except ValueError:
  27.             print("無効な数字です。") #範囲から逸脱した数字を入力したときに出る。
  29. if __name__ == "__main__": #スクリプトが直接実行された場合に内部のコードを実行するための条件と出てきたが、何のことだか全くわからない。このコードが無くても一応動いた。
  30.     main()
